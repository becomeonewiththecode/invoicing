# Database diagram

```mermaid
erDiagram
  users ||--o{ clients : owns
  users ||--o{ discount_codes : owns
  users ||--o{ support_tickets : submits
  users ||--o{ content_flags : flagged
  users ||--o{ backup_snapshots : has
  users ||--o{ backup_policies : has
  clients ||--o{ invoices : has
  clients ||--o{ projects : has
  users ||--o{ projects : owns
  projects ||--o{ project_external_links : has
  projects ||--o{ invoices : "optional FK project_id"
  invoices ||--o{ invoice_items : contains
  invoices ||--o{ payment_reminders : logs
  support_tickets ||--o{ ticket_messages : has

  users {
    uuid id PK
    string email UK
    string password_hash
    string business_name
    string business_address
    string business_phone
    string business_email
    string tax_id
    decimal default_hourly_rate
    decimal default_tax_rate
    string business_website
    string business_fax
    string logo_url
    string smtp_host
    int smtp_port
    string smtp_user
    string smtp_pass
    string smtp_from
    int client_counter
    text payable_text
    string role "user or admin"
    timestamptz created_at
    timestamptz updated_at
  }

  clients {
    uuid id PK
    uuid user_id FK
    string customer_number
    string name
    string email
    string phone
    string company
    string address
    text notes
    string discount_code
    boolean portal_enabled
    string portal_login_email "optional login email (unique)"
    string portal_token "unique access token (generated by vendor)"
    string portal_password_hash "bcrypt hash (required when enabled)"
    string portal_totp_secret "TOTP secret (set on setup)"
    boolean portal_totp_enabled
    timestamptz created_at
    timestamptz updated_at
  }

  projects {
    uuid id PK
    uuid client_id FK
    uuid user_id FK
    string name
    text description
    string status
    string priority
    jsonb milestones
    timestamptz created_at
    timestamptz updated_at
  }

  project_external_links {
    uuid id PK
    uuid project_id FK
    text url
    text description
    int sort_order
    timestamptz created_at
  }

  invoices {
    uuid id PK
    uuid user_id FK
    uuid client_id FK
    uuid project_id FK "nullable ON DELETE SET NULL"
    string invoice_number
    invoice_status status "draft sent paid late cancelled"
    date issue_date
    date due_date
    decimal subtotal
    decimal tax_rate
    decimal tax_amount
    string discount_code
    decimal discount_amount
    decimal total
    text notes
    boolean is_recurring
    string recurrence_interval
    date next_recurrence_date
    timestamptz sent_at "sent late logic"
    string share_token "nullable unique public link"
    timestamptz created_at
    timestamptz updated_at
  }

  invoice_items {
    uuid id PK
    uuid invoice_id FK
    string description
    decimal quantity
    decimal unit_price
    decimal amount
    int sort_order
    timestamptz created_at
  }

  discount_codes {
    uuid id PK
    uuid user_id FK
    string code
    string description
    string type "percent or fixed"
    decimal value
    boolean is_active
    timestamptz created_at
  }

  payment_reminders {
    uuid id PK
    uuid invoice_id FK
    timestamptz sent_at "reminder log time"
    string reminder_type
  }

  support_tickets {
    uuid id PK
    uuid user_id FK
    string subject
    string status "open in_progress closed"
    string priority "low normal high urgent"
    timestamptz created_at
    timestamptz updated_at
  }

  ticket_messages {
    uuid id PK
    uuid ticket_id FK
    uuid sender_id FK
    text body
    boolean is_admin_reply
    timestamptz created_at
  }

  content_flags {
    uuid id PK
    uuid user_id FK
    string content_type
    text content_snippet
    string reason
    string status "pending approved rejected"
    uuid reviewed_by FK
    timestamptz reviewed_at
    timestamptz created_at
  }

  backup_snapshots {
    uuid id PK
    uuid user_id FK
    text file_path
    bigint file_size_bytes
    boolean is_automated
    boolean verified
    timestamptz verified_at
    timestamptz created_at
  }

  backup_policies {
    uuid id PK
    uuid user_id FK
    int retention_days
    int max_snapshots
    boolean is_enabled
    string cron_expression
    timestamptz created_at
    timestamptz updated_at
  }

  system_logs {
    uuid id PK
    string level
    string source
    string method
    text path
    int status_code
    int response_time_ms
    string ip
    uuid user_id
    text error_message
    jsonb metadata
    timestamptz created_at
  }

  rate_limit_configs {
    uuid id PK
    string route_pattern UK
    int window_ms
    int max_requests
    boolean is_enabled
    timestamptz updated_at
  }

  rate_limit_events {
    uuid id PK
    string ip
    text path
    boolean was_blocked
    timestamptz created_at
  }
```

Mermaid `erDiagram` is supported on GitHub and many Markdown viewers; for strict PostgreSQL types, column defaults, and indexes, see [schema.md](schema.md).

**Notes**

- **`invoices.project_id`** is nullable (optional related project); the crow’s foot from `projects` to `invoices` reflects invoices that *may* reference a project.
- `payment_reminders.sent_at` is the time a reminder was logged, not the same field as `invoices.sent_at` (when the invoice was marked sent).
